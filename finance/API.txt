//
// API - набор функций, предоставляемых данным плагином
//

1. finance_check_money($userlogin, $balance_no = -1) 
   :: проверка баланса пользователя (в единицах валюты)
   $userlogin  - login пользователя чей баланс проверять
   $balance_no - номер баланса для проверки (0 - основной баланс, -1 - считать 
                 общую сумму со всех монетарных балансов)
   Результат: Возвращает сумму на счету пользователя.


2. finance_check_enough_money($userlogin, $type, $tprice, $price)
   :: Проверка доступности у пользователя средств для определённого типа действий
   $userlogin   - логин пользователя
   $type	- тип действия
   $tprice	- цена в "единицах" при наличии баланса по данному типу
   $price	- цена в валюте
   Результат: Возвращает 0 если денег недостаточно и 1 в противном случае


3. finance_get_balance_type($balance_no)
   :: Получить тип баланса
   $balance_no	- Номер баланса
   Результат: Пустая строка в случае монетарного баланса и текст-тип если баланс
              немонетарный


4. finance_pay($identity, $payment)
   :: Снятие суммы с баланса пользователя
   $identity - массив идентифицирующий пользователя. Заполняется один из параметров
  	# 'id'	- ID пользователя
  	# 'login'	- Login пользователя
   $payment - массив с параметрами платежа
    # 'type' - тип проводимого платежа:
  	'money'  -	_только_ валютный
  	'points' -	_только_ в поинтах
  	'auto'   -	[default] сначала пытаемся списать в поинтах, если их недостаточно -
  					 списываем средства в валюте
    # 'ptype' - тип поинтов для оплаты (в случае авто оплаты или оплаты по поинтам)
    # 'value' - массив с параметрами платежа
  	'money'  =	сумма платежа в валюте
  	'points' =	массив (тип поинтов,кол-во поинтов) при списании в поинтах
    # 'description' - описание платежа [ для помещения в таблицу finance_history ]
  

   Результат: 0 в случае неуспеха, 1 в случае успеха
